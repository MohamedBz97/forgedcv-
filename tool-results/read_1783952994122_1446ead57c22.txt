  1726→export function getAllPosts(): BlogPost[] {

  1733→export function getRelatedPosts(slug: string, n = 3): BlogPost[] {
  1734→  const current = getPostBySlug(slug);
  1735→  if (!current) return [];
  1736→  const sameCategory = BLOG_POSTS.filter(
  1737→    (post) => post.slug !== slug && post.category === current.category,
  1738→  );
  1739→  if (sameCategory.length >= n) return sameCategory.slice(0, n);
  1740→  // Backfill with posts from other categories, excluding self.
  1741→  const others = BLOG_POSTS.filter(
  1742→    (post) => post.slug !== slug && post.category !== current.category,
  1743→  );
  1744→  return [...sameCategory, ...others].slice(0, n);
  1745→}
